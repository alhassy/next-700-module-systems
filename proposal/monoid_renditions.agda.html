record Monoid-Record : Set₁ where
  infixl 5 _⨾_
  field
    -- Interface

-- Type alias
Monoid-Σ  :  Set₁
Monoid-Σ  =    Σ Carrier ∶ Set
	     • Σ Id ∶ Carrier
	     • Σ _⨾_ ∶ (Carrier → Carrier → Carrier)
	     • Σ lid ∶ (∀{x} → Id ⨾ x ≡ x)
	     • Σ rid ∶ (∀{x} → x ⨾ Id ≡ x)
	     • (∀ x y z → (x ⨾ y) ⨾ z ≡ x ⨾ (y ⨾ z))

pop-Id-Σ : ∀{{M : Monoid-Σ}}
		       (let Id  = proj₁ (proj₂ M))
		       (let _⨾_ = proj₁ (proj₂ (proj₂ M)))
		   →  ∀ (x y : proj₁ M)  →  (x ⨾ Id) ⨾ y  ≡  x ⨾ y

		     where  _⨾_    = proj₁ (proj₂ (proj₂ M))
			    rid    = proj₁ (proj₂ (proj₂ (proj₂ (proj₂ M))))
